Codice SQL per Supabase

-- Crea la tabella per le tracce audio
CREATE TABLE tracce_audio (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  titolo VARCHAR(255) NOT NULL,
  artista VARCHAR(255) NOT NULL,
  genere VARCHAR(50) NOT NULL,
  url TEXT NOT NULL,
  durata INTEGER NOT NULL, -- in secondi
  dimensione BIGINT NOT NULL, -- in bytes
  data_caricamento TIMESTAMP DEFAULT now()
);

-- Crea un indice per migliorare le performance di ricerca
CREATE INDEX idx_tracce_genere ON tracce_audio(genere);
CREATE INDEX idx_tracce_titolo ON tracce_audio(titolo);
CREATE INDEX idx_tracce_artista ON tracce_audio(artista);

-- Crea il bucket di storage per i file audio
INSERT INTO storage.buckets (id, name, public) VALUES (
  'audio_files', 'audio_files', true
);

-- Policy per permettere a tutti di leggere i file audio
CREATE POLICY "Audio pubblico" ON storage.objects
FOR SELECT USING (bucket_id = 'audio_files');

-- Policy per permettere agli utenti autenticati di caricare file
CREATE POLICY "Caricamento audio" ON storage.objects
FOR INSERT WITH CHECK (
  bucket_id = 'audio_files' AND
  (auth.role() = 'authenticated')
);
